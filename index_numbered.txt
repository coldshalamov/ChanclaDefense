   613	            function drawChainEffects() {
   614	                chainEffects.forEach(eff => {
   615	                    ctx.save();
   616	                    const p = eff.timer / eff.maxTime;
   617	                    ctx.globalAlpha = p;
   618	                    // Cyan electricity glow
   619	                    ctx.shadowColor = '#00e5ff';
   620	                    ctx.shadowBlur = 10;
   621	                    ctx.strokeStyle = '#00e5ff';
   622	                    ctx.lineWidth = 3;
   623	                    ctx.lineCap = 'round';
   624	                    ctx.lineJoin = 'round';
   625
   626	                    const segments = 6;
   627	                    ctx.beginPath();
   628	                    ctx.moveTo(eff.x1, eff.y1);
   629
   630	                    const dx = (eff.x2 - eff.x1) / segments;
   631	                    const dy = (eff.y2 - eff.y1) / segments;
   632
   633	                    // Vector for jitter direction (perpendicular)
   634	                    const len = Math.sqrt(dx * dx + dy * dy);
   635	                    const nx = -dy / (len || 1);
   636	                    const ny = dx / (len || 1);
   637
   638	                    for (let i = 1; i < segments; i++) {
   639	                        // More jitter in the middle, less at ends
   640	                        const midFactor = 1 - Math.abs(i - segments / 2) / (segments / 2);
   641	                        const jitter = (Math.random() - 0.5) * 40 * midFactor;
   642	                        const tx = eff.x1 + dx * i + nx * jitter;
   643	                        const ty = eff.y1 + dy * i + ny * jitter;
   644	                        ctx.lineTo(tx, ty);
   645	                    }
   646	                    ctx.lineTo(eff.x2, eff.y2);
   647	                    ctx.stroke();
   648
   649	                    // White hot core
   650	                    ctx.strokeStyle = '#fff';
   651	                    ctx.lineWidth = 1.5;
   652	                    ctx.shadowBlur = 0;
   653	                    ctx.stroke();
   654
   655	                    ctx.restore();
   656	                });
   657	            }

  1330	            function wrapText(context, text, x, y, maxWidth, lineHeight) {
  1331	                const words = text.split(' ');
  1332	                let line = '';
  1333	                let lines = [];
  1334	                for (let n = 0; n < words.length; n++) {
  1335	                    const test = line + words[n] + ' ';
  1336	                    const metrics = context.measureText(test);
  1337	                    if (metrics.width > maxWidth && n > 0) {
  1338	                        lines.push(line);
  1339	                        line = words[n] + ' ';
  1340	                    } else {
  1341	                        line = test;
  1342	                    }
  1343	                }
  1344	                lines.push(line);
  1345	                lines.forEach((l, i) => context.fillText(l.trim(), x, y + i * lineHeight - (lines.length - 1) * lineHeight / 2));
  1346	            }
